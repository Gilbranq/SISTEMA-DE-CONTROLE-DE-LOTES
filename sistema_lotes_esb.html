SQL da view
CREATE OR REPLACE VIEW VW_ALERTAS_VALIDADE AS
SELECT
  l.ID_LOTE, l.NUMERO_LOTE, l.PRODUTO_DESCRICAO,
  l.DATA_VALIDADE, l.QUANTIDADE,
  CASE
    WHEN l.DATA_VALIDADE < SYSDATE THEN 'CRÍTICO - VENCIDO'
    WHEN l.DATA_VALIDADE <= SYSDATE + 7 THEN 'URGENTE - 7 DIAS'
    WHEN l.DATA_VALIDADE <= SYSDATE + 15 THEN 'ALERTA - 15 DIAS'
    WHEN l.DATA_VALIDADE <= SYSDATE + 30 THEN 'ATENÇÃO - 30 DIAS'
  END AS NIVEL_ALERTA
FROM LOTE_CONTROLE l
WHERE l.STATUS = 'ATIVO'
  AND l.DATA_VALIDADE <= SYSDATE + 30
ORDER BY l.DATA_VALIDADE ASC;

Trigger PL/SQL
CREATE OR REPLACE TRIGGER TRG_LOTE_VALIDACAO
BEFORE INSERT OR UPDATE ON LOTE_CONTROLE
FOR EACH ROW
BEGIN
  -- RN001: Validação de datas
  IF :NEW.DATA_FABRICACAO >= :NEW.DATA_VALIDADE THEN
    RAISE_APPLICATION_ERROR(-20001,
      'Data de fabricação deve ser anterior à data de validade');
  END IF;

  -- RN002: Validação de quantidade
  IF :NEW.QUANTIDADE <= 0 THEN
    RAISE_APPLICATION_ERROR(-20002,
      'Quantidade do lote deve ser maior que zero');
  END IF;

  -- Auto-atualiza status se vencido
  IF :NEW.DATA_VALIDADE < SYSDATE AND :NEW.STATUS = 'ATIVO' THEN
    :NEW.STATUS := 'VENCIDO';
  END IF;
END;
